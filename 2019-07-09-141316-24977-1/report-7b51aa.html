102void TelsemAtlas::equare()
103{
104 Index maxlat = static_cast<Index>(180.0 / dlat);
105
106 ncells.resize(maxlat);
107 firstcells.resize(maxlat);
108
109 // Total number of cells.
110 Index totcel = 0.0;
111
112 Numeric rcelat = dlat * PI / 180.0;
113
114 Numeric hezon = EARTH_RADIUS * sin(rcelat);
115 Numeric aezon = 2.0 * PI * EARTH_RADIUS * hezon;
116 Numeric aecell = aezon * dlat / 360.0;
117
118 for (Index i = 0; i < maxlat / 2; ++i) {
119 Numeric xlatb = static_cast<Numeric>(i) * dlat;
120 Numeric xlate = xlatb + dlat;
121 Numeric rlatb = DEG2RAD * xlatb;
122 Numeric rlate = DEG2RAD * xlate;
123 Numeric htb = EARTH_RADIUS * sin(rlatb);
124 Numeric hte = EARTH_RADIUS * sin(rlate);
125 Numeric htzone = hte - htb;
126 Numeric azone = 2.0 * PI * EARTH_RADIUS * htzone;
127 Numeric rcells = azone / aecell;
128 Index icellr = static_cast<Index>(rcells + 0.5);
129
130 totcel += 2 * icellr;
131
132 Index lat1 = i + maxlat / 2;
133 Index lat2 = maxlat / 2 - 1 - i;
134 ncells[lat1] = icellr;
135 ncells[lat2] = icellr;
136 }
137
138 firstcells[0] = 0;
139 for (Index i = 1; i < maxlat; ++i) {
140 firstcells[i] = firstcells[i - 1] + ncells[i];
141 }
142}

154Index TelsemAtlas::calc_cellnum(Numeric lat,
155 Numeric lon) const
156{
157 if ((lat < -90.0) || (lat > 90.0)) {
158 throw std::runtime_error("Latitude input must be within the range [-90.0, 90.0].");
159 }
160
161 if ((lon < 0.0) || (lon > 360.0)) {
162 throw std::runtime_error("Longitude input must be within the range [0.0, 360.0].");
163 }
164
165 // Avoid corner cases that hit the outermost edge of the atlas cells.
166 if (lat == 90.0) {
167 lat -= 0.125;
168 }
169
170 if (lon == 360.0) {
171 lat -= 0.125;
172 }
173
174 Index cellnum = 0;
175 Index ilat = static_cast<Index>((lat + 90.0) / dlat);
176 Index ilon = static_cast<Index>(lon / (360.0 / static_cast<Numeric>(ncells[ilat]))) + 1;
177 for (Index i = 0; i < ilat; ++i) {
178 cellnum += ncells[i];
179 }
180 cellnum += ilon;
181 return cellnum;
182}

240std::pair<Numeric, Numeric> TelsemAtlas::get_coordinates(Index cellnum) const
241{
242
243 Index index_lat_max = static_cast<Index>(180.0 / dlat);
244 Index index_lat = -1;
245 Index index_lon = -1;
246 if (cellnum >= firstcells[index_lat_max-1]) {
247 index_lat = index_lat_max;
248 index_lon = cellnum - firstcells[index_lat_max];
249 } else {
250 for (Index i = 0; i < index_lat_max; ++i) {
251 if ((cellnum >= firstcells[i]) && (cellnum < firstcells[i + 1])) {
252 index_lat = i;
253 index_lon = cellnum - firstcells[i];
254 }
255 }
256 }
257 Numeric lat = (static_cast<Numeric>(index_lat) - 0.5) * dlat - 90.0;
258 Numeric lon = (static_cast<Numeric>(index_lon) - 0.5)
259 * (360.0 / static_cast<Numeric>(ncells[index_lat]));
260 return std::make_pair(lat, lon);
261}

313std::pair<Numeric, Numeric> TelsemAtlas::emis_interp(Numeric theta,
314 Numeric freq,
315 Index class1,
316 Index class2,
317 const ConstVectorView &ev,
318 const ConstVectorView &eh) const
319{
320 Vector emiss_scal_h(3);
321 Vector emiss_scal_v(3);
322
323 for (Index i = 0; i < 3; ++i) {
324 Numeric e0 = a0_k0[i + (class1 - 1) * 3]
325 + a0_k1[i + (class1 - 1) * 3] * ev[i]
326 + a0_k2[i + (class1 - 1) * 3] * eh[i];
327
328 Numeric a0 = a0_eveh[i + (class1 - 1) * 3];
329 Numeric a1 = a1_eveh[i + (class1 - 1) * 3];
330 Numeric a2 = a2_eveh[i + (class1 - 1) * 3];
331 Numeric a3 = a3_eveh[i + (class1 - 1) * 3];
332 Numeric b0 = b0_eveh[i + (class1 - 1) * 3];
333 Numeric b1 = b1_eveh[i + (class1 - 1) * 3];
334 Numeric b2 = b2_eveh[i + (class1 - 1) * 3];
335 Numeric b3 = b3_eveh[i + (class1 - 1) * 3];
336
337 Numeric s1_v = (theta - 53.0) / -53.0 * (e0 - a0) / a0;
338 Numeric em53_v = a3 * pow(53.0, 3.0) + a2 * pow(53.0, 2.0) + a1 * 53.0 + a0;
339 Numeric s2_v = theta / 53.0 * (ev[i] - em53_v) / em53_v;
340 Numeric s_v = 1.0 + s1_v + s2_v;
341
342 Numeric emtheta_v = a3 * pow(theta, 3) + a2 * pow(theta, 2) + a1 * theta + a0;
343 emiss_scal_v[i] = s_v * emtheta_v;
344
345 Numeric s1_h = (theta - 53.0) / -53.0 * (e0 - b0) / b0;
346 Numeric em53_h = b3 * pow(53.0, 3.0) + b2 * pow(53.0, 2.0) + b1 * 53.0 + b0;
347 Numeric s2_h = theta / 53.0 * (eh[i] - em53_h) / em53_h;
348 Numeric s_h = 1.0 + s1_h + s2_h;
349
350 Numeric emtheta_h = b3 * pow(theta, 3) + b2 * pow(theta, 2) + b1 * theta + b0;
351 emiss_scal_h[i] = s_h * emtheta_h;
352
353 }
354
355 Numeric emiss_h = interp_freq2(emiss_scal_h[0],
356 emiss_scal_h[1],
357 emiss_scal_h[2],
358 freq,
359 class2);
360 Numeric emiss_v = interp_freq2(emiss_scal_v[0],
361 emiss_scal_v[1],
362 emiss_scal_v[2],
363 freq,
364 class2);
365
366 if (emiss_v < emiss_h) {
367 emiss_v = 0.5 * (emiss_v + emiss_h);
368 emiss_h = emiss_v;
369 }
370 return std::make_pair(emiss_v, emiss_h);
371}
