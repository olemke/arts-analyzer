890inline void calculate_G_from_relmat(Vector& G, const ArrayOfLineRecord& lines, const ConstMatrixView W)
891{
892 const Index nl = lines.nelem();
893
894 Vector d0(nl);
895 for(Index il=0; il<nl; il++)
896 d0[il] = reduced_dipole(lines[il]);
897
898 G = Vector(nl, 0);
899 for(Index il1=0; il1<nl; il1++) {
900 Numeric a=0, b=0, c=0, d=0;
901 Numeric df, r;
902 for(Index il2=0; il2<nl; il2++) {
903 df = lines[il1].F() - lines[il2].F();
904 r = d0[il2] / d0[il1];
905 if(il1 not_eq il2) {
906 a += W(il1, il2) * W(il2, il1) / (df*df);
907 b += r * W(il2, il1) / df;
908 c += r * W(il1, il2) * W(il1, il1) / (df*df);
909 for(Index il3=0; il3<nl; il3++) {
910 if(il3 not_eq il2) {
911 d += r * W(il2, il3) * W(il3, il1) / (-df * (lines[il3].F() - lines[il1].F()));
912 }
913 }
914 }
915 }
916 G[il1] = a - b*b + 2*c - 2*d;
917 }

1944 const Index nline = abs_lines_per_band[iband].nelem();
1945
1946 for(Index iline = 0; iline < nline; iline++) {
1947 for(Index ilevl = 0; ilevl < nlevl; ilevl++)
1948 for(Index idata = 0; idata < ndata; idata++)
1949 data[idata][ilevl] = relmat_per_band[ilevl][iband](idata, iline);
1950
1951 // data vector for holding the results before setting the line
1952 Vector dvec(10, 0);
1953
1954 /* dvec structure:
1955 * 0 : y0
1956 * 1 : y1
1957 * 2 : g0
1958 * 3 : g1
1959 * 4 : d0
1960 * 5 : d1
1961 * 6 : T0
1962 * 7 : n
1963 * 8 : 2*n
1964 * 9 : 2*n
1965 *
1966 * Note that while all n and T0 are known from the line catalog, for legacy purpose this methods needs to set them
1967 * These legacy reasons are bad to have here because they might create the confusion that you can change these and
1968 * still be OK with the calculations... regardless though, the method using these numbers is worse than direct
1969 * line mixing so some errors should be expected
1970 *
1971 * Inteded computations:
1972 *
1973 * Fit data to
1974 *
1975 * X = (F0 + F1 (T0/T - 1)) ((T0 / T)^n * P)^k,
1976 *
1977 * where F is any of y, g, d above, and T is temperature, and P is the pressure. k is 1 for y but 2 for the
1978 * others. n is the air pressure broadening parameter (FIXME: generalize to other planets)
1979 */
1980
1981 const Numeric T0 = abs_lines_per_band[iband][iline].Ti0();
1982 const Numeric n = abs_lines_per_band[iband][iline].PressureBroadeningAirBroadeningNair();
1983
1984 dvec[6] = T0;
1985 dvec[7] = n;
1986 dvec[8] = 2*n;
1987 dvec[9] = 2*n;
1988
1989 // Fitting parameters for computations solving A(T, P) dC = data(P, T)-f(C, P, T); C += dC;
1990 Vector C(2), dC(2);
1991 Matrix A(nlevl, 2);
1992
1993 const static Index max_loop_count = 10;
1994 const static Numeric res_limit = 1e-30;
1995 Index pos = 0;
1996 bool squared = false;
1997
1998 // Do similar fitting for all data vectors
1999 for(auto& v : data) {
2000 // Take a value from the center of the data and use it as a starting point for the fitting
2001 C[0] = v[nlevl/2] / rtp_pressure / (squared ? rtp_pressure : 1.0);
2002 C[1] = v[nlevl/2] / rtp_pressure / (squared ? rtp_pressure : 1.0);
2003
2004 Numeric res;
2005 Index loop_count=0;
2006 do {
2007 for(Index ilevl = 0; ilevl < nlevl; ilevl++) {
2008 const Numeric theta = T0 / abs_t[ilevl];
2009 const Numeric theta_n = pow(theta, n);
2010 const Numeric TP = theta_n * rtp_pressure * (squared ? theta_n * rtp_pressure : 1.0);
2011 A(ilevl, 0) = TP;
2012 A(ilevl, 1) = (theta - 1.0) * TP;
2013 const Numeric f = C[0] * TP + C[1] * (theta - 1.0) * TP;
2014 delta[ilevl] = v[ilevl] - f;
2015 }
2016 res = lsf(dC, A, delta);
2017 C += dC;
2018 loop_count++;
2019 } while(res > res_limit and loop_count < max_loop_count);
2020
2021 dvec[pos] = C[0]; pos++;
2022 dvec[pos] = C[1]; pos++;
2023
2024 // Only the first loop is not squared
2025 squared = true;
2026 }
2027
2028 // Overwrite the linemixing data NOTE: Conversion to modern LM-format... y0 y1 ny, g0...dv1 ndv
2029 abs_lines_per_band[iband][iline].SetLineMixing2SecondOrderData(
2030 Vector({dvec[0], dvec[1], dvec[7],
2031 dvec[2], dvec[3], dvec[8],
2032 dvec[4], dvec[5], dvec[9]}));
2033 }
2034 }
